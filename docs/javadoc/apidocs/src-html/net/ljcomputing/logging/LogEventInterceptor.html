017package net.ljcomputing.logging;
018
019import net.ljcomputing.gson.converter.GsonConverterService;
020import net.ljcomputing.logging.annotation.LogEvent;
021
022import org.slf4j.Logger;
023import org.slf4j.LoggerFactory;
024
025import org.springframework.beans.factory.annotation.Autowired;
026import org.springframework.stereotype.Component;
027
028import org.aspectj.lang.JoinPoint;
029import org.aspectj.lang.annotation.AfterReturning;
030import org.aspectj.lang.annotation.Aspect;
031import org.aspectj.lang.annotation.Before;
032
033/**
034 * Log event interceptor.
035 * 
036 * @author James G. Willmore
037 *
038 */
039@Aspect
040@Component
041public class LogEventInterceptor {
042
043  /** The Gson converter service. */
044  @Autowired
045  private GsonConverterService gsonConverterService;
046
047  /**
048   * Log an event.
049   *
050   * @param jp
051   *            the join point
052   * @param bean
053   *            the bean
054   * @param event
055   *            the event
056   */
057  @Before(value = "net.ljcomputing.logging.LogEventManager.logEvent()"
058      + "&& target(bean) "
059      + "&& @annotation(net.ljcomputing.logging.annotation.LogEvent)"
060      + "&& @annotation(event)", argNames = "bean,event")
061  public void log(JoinPoint jp, Object bean, LogEvent event) {
062    LogEvent.Level level = event.level();
063    String message = event.message();
064    String method = jp.getSignature().getName();
065    Logger logger = LoggerFactory.getLogger(bean.getClass());
066
067    log(logger, level, "Method: " + method + " - " + message);
068
069    if(event.showArgs()) {
070      log(logger, level, "     arguments passed:");
071      Object[] signatureArgs = jp.getArgs();
072      for(Object arg : signatureArgs) {
073        log(logger, level, "      arg : " + arg);
074      }
075    }
076
077  }
078
079  /**
080   * Log response.
081   *
082   * @param jp the jp
083   * @param bean the bean
084   */
085  @AfterReturning(
086      value = "net.ljcomputing.logging.LogEventManager.logResponse()"
087          + "&& @annotation(net.ljcomputing.logging.annotation.LogResponse)",
088      returning = "bean")
089  public void logResponse(JoinPoint jp, Object bean) {
090    String method = jp.getSignature().getName();
091    Logger logger = LoggerFactory.getLogger(bean.getClass());
092
093    try {
094      String result = gsonConverterService.toJson(bean);
095      logger.debug("Method: " + method + "==> " + result + " <==");
096    } catch (Exception e) {
097      logger.error("Failed to log a response: {}", e);
098    }
099  }
100
101  /**
102   * Log event based upon the bean's class, given log event level, and
103   * message.
104   *
105   * @param logger
106   *            the logger
107   * @param level
108   *            the level
109   * @param message
110   *            the message
111   */
112  private static void log(Logger logger, LogEvent.Level level, String message) {
113    switch (level) {
114      case TRACE:
115        logger.trace(message);
116        break;
117      case DEBUG:
118        logger.debug(message);
119        break;
120      case INFO:
121        logger.info(message);
122        break;
123      case WARN:
124        logger.warn(message);
125        break;
126      case ERROR:
127        logger.error(message);
128        break;
129    }
130  }
131
132}
